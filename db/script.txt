CREATE TABLE comment (
    comment_id     BIGINT UNSIGNED AUTO_INCREMENT NOT NULL,
    pet_id         BIGINT UNSIGNED NOT NULL,
    post_id        BIGINT UNSIGNED NOT NULL,
    content        VARCHAR(140) NOT NULL,
    date           DATETIME NOT NULL,
    active         BOOL NOT NULL,
    PRIMARY KEY ( comment_id )
);

CREATE TABLE event (
    event_id      BIGINT UNSIGNED AUTO_INCREMENT NOT NULL,
    pet_id        BIGINT UNSIGNED NOT NULL,
    name          VARCHAR(100) NOT NULL,
    description   VARCHAR(280) NOT NULL,
    latitude      DOUBLE NOT NULL,
    longitude     DOUBLE NOT NULL,
    active        BOOL NOT NULL,
    PRIMARY KEY ( event_id )
);

CREATE TABLE follower (
    follower_id    BIGINT UNSIGNED AUTO_INCREMENT NOT NULL,
    pet_follower   BIGINT UNSIGNED NOT NULL,
    pet_followed   BIGINT UNSIGNED NOT NULL,
    active         BOOL NOT NULL,
    PRIMARY KEY ( follower_id )
);

CREATE TABLE enjoy (
    enjoy_id  BIGINT UNSIGNED AUTO_INCREMENT NOT NULL,
    pet_id    BIGINT UNSIGNED NOT NULL,
    post_id   BIGINT UNSIGNED NOT NULL,
    active    BOOL NOT NULL,
    PRIMARY KEY ( enjoy_id )
); 

CREATE TABLE notification (
    notification_id   BIGINT UNSIGNED AUTO_INCREMENT NOT NULL,
    notified_pet_id   BIGINT UNSIGNED NOT NULL,
    recived_pet_id    BIGINT UNSIGNED NOT NULL,
    message           VARCHAR(150) NOT NULL,
    active            BOOL NOT NULL,
	PRIMARY KEY ( notification_id )
);

CREATE TABLE pet (
    pet_id        BIGINT UNSIGNED AUTO_INCREMENT NOT NULL,
    user_id       BIGINT UNSIGNED NOT NULL,
    name          VARCHAR(50) NOT NULL,
    species       VARCHAR(30) NOT NULL,
    description   VARCHAR(280),
    genre         VARCHAR(10) NOT NULL,
    photo         VARCHAR(1000) NOT NULL,
    active        BOOL NOT NULL,
    PRIMARY KEY ( pet_id )
);

CREATE TABLE post (
    post_id       BIGINT UNSIGNED AUTO_INCREMENT NOT NULL,
    pet_id        BIGINT UNSIGNED NOT NULL,
    comment_id    BIGINT UNSIGNED NOT NULL,
    enjoy_id      BIGINT UNSIGNED NOT NULL,
    photos        VARCHAR(1000),
    description   VARCHAR(280) NOT NULL,
    DATE          DATETIME NOT NULL,
    active        BOOL NOT NULL,
	PRIMARY KEY ( post_id )
);

CREATE TABLE user (
    user_id    BIGINT UNSIGNED AUTO_INCREMENT NOT NULL,
    name       VARCHAR(100) NOT NULL,
    nickname   VARCHAR(50) NOT NULL,
    email      VARCHAR(100) NOT NULL,
    password   VARCHAR(80) NOT NULL,
    photo      VARCHAR(1000),
    active     BOOL NOT NULL,
    PRIMARY KEY ( user_id )
);
